HASH_SET_ACTOR_NAME_TEMPLATE = "HashSetActor_{dedup_uid}_{index}"


@ray.remote
class HashSetActor:
    def __init__(self, parallel_id: int, parallel_num: int):
        self.parallel_id = parallel_id
        self.parallel_num = parallel_num
        self.seen: set[bytes] = set()

    def add_and_check(self, hash_list: list[tuple[bytes, int]]) -> list[int]:
        dup_indexes: list[int] = []
        for hash_val, idx in hash_list:
            if hash_val in self.seen:
                dup_indexes.append(idx)
            else:
                self.seen.add(hash_val)
        return dup_indexes

class HashDedup:
    """
    Exact-hash-based deduplication for text, binary data, or files.

    Computes a hash for each record and uses distributed HashSet Actors to detect
    duplicates. The first occurrence of each hash value is kept; subsequent duplicates
    are filtered out.

    Args:
        hash_algorithm: Hash algorithm to use. Accepts any algorithm supported by
            :mod:`hashlib` (e.g. ``'md5'``, ``'sha1'``, ``'sha256'``, ``'sha3_256'``,
            ``'blake2b'``) or :mod:`xxhash` (e.g. ``'xxh64'``, ``'xxh3_64'``,
            ``'xxh3_128'``). Default: ``'xxh3_128'``.
        parallel_num: Number of HashSet Actor shards. If ``None``, it is derived
            automatically from the cluster CPU count (``max(cpu_count // 4, 1)``).
        is_file: If ``True``, treat each column value as a file path supported by
            `fsspec <https://filesystem-spec.readthedocs.io>`_ (e.g. local paths,
            ``s3://``, ``cos://``, ``gs://``, etc.) and hash the file contents.
            Useful for deduplicating audio, video, or other binary files by content.
            Default: ``False``.

    Example — text deduplication::

        >>> from xpark.dataset import read_parquet
        >>> from xpark.dataset.filters.dedup import HashDedup
        >>> from xpark.dataset.expressions import col
        >>>
        >>> ds = read_parquet("/data/my-dataset")
        >>> ds.filter(
        ...     HashDedup().with_column(text=col("text"))
        ... ).write_parquet("/data/dedup-my-dataset")

    Example — video file deduplication::

        >>> from xpark.dataset import read_parquet
        >>> from xpark.dataset.filters.dedup import HashDedup
        >>> from xpark.dataset.expressions import col
        >>>
        >>> # Each row has a "video_path" column pointing to a local video file.
        >>> ds = read_parquet("/data/video-dataset")
        >>> ds.filter(
        ...     HashDedup(hash_algorithm="sha256", is_file=True).with_column(text=col("video_path"))
        ... ).write_parquet("/data/dedup-video-dataset")
    """

    def __init__(
        self,
        hash_algorithm: str = "xxh3_128",
        parallel_num: int | None = None,
        is_file: bool = False,
    ):
        self.hash_algorithm = hash_algorithm
        self.is_file = is_file

        # Validate the algorithm early so users get a clear error at construction time.
        if (
            self.hash_algorithm not in xxhash.algorithms_available
            and self.hash_algorithm not in hashlib.algorithms_available
        ):
            raise ValueError(
                f"Unsupported hash algorithm: '{self.hash_algorithm}'. "
                f"Supported algorithms: xxhash={sorted(xxhash.algorithms_available)}, "
                f"hashlib={sorted(hashlib.algorithms_available)}"
            )

        if parallel_num is None:
            self.parallel_num = max(int(ray.cluster_resources().get("CPU", 1) / 4), 1)
        else:
            self.parallel_num = int(parallel_num)

        assert self.parallel_num > 0

        dedup_uid = uuid.uuid4().hex[:8]
        logger.info(
            "HashDedup[%s] parallel_num: %s, hash_algorithm: %s, is_file: %s",
            dedup_uid,
            self.parallel_num,
            self.hash_algorithm,
            self.is_file,
        )

        self.hash_set_actors = [
            HashSetActor.options(
                name=HASH_SET_ACTOR_NAME_TEMPLATE.format(dedup_uid=dedup_uid, index=index),
            ).remote(
                parallel_id=index,
                parallel_num=self.parallel_num,
            )
            for index in range(self.parallel_num)
        ]

    def _process(
        self,
        text_column: ColumnExpr,
        dataset: ray.data.Dataset,
        compute: ComputeStrategy | None = None,
        **ray_remote_args,
    ) -> ray.data.Dataset:
        hash_algorithm = self.hash_algorithm
        parallel_num = self.parallel_num
        hash_set_actors = self.hash_set_actors
        is_file = self.is_file

        @udf(return_dtype=pa.bool_())
        def _filter_dup(array: pa.ChunkedArray) -> pa.Array:

            shard_queries: dict[int, list[tuple[bytes, int]]] = {i: [] for i in range(parallel_num)}

            async def _compute_all() -> None:
                tasks = []
                indices = []
                for idx, scalar in enumerate(array):
                    value = scalar.as_py()
                    if value is None:
                        # None values are kept as-is (not deduplicated).
                        continue
                    tasks.append(compute_hash(value, is_file=is_file, algorithm=hash_algorithm))
                    indices.append(idx)

                hashes = await asyncio.gather(*tasks)
                for h, idx in zip(hashes, indices):
                    shard_id = int.from_bytes(h[:8], byteorder="big") % parallel_num
                    shard_queries[shard_id].append((h, idx))

            asyncio.run(_compute_all())

            # Query each shard Actor in parallel.
            result_refs = [
                hash_set_actors[shard_id].add_and_check.remote(queries)
                for shard_id, queries in shard_queries.items()
                if queries
            ]
            dup_index_lists = ray.get(result_refs)

            # Build mask: True = keep, False = duplicate to be filtered out.
            mask = np.ones(len(array), dtype=np.bool_)
            for dup_indexes in dup_index_lists:
                for idx in dup_indexes:
                    mask[idx] = False

            return pa.array(mask)

        return dataset.filter(expr=_filter_dup(text_column), compute=compute, **ray_remote_args)  # type: ignore[arg-type]

class HashDedupOp(DedupOp):
    def __init__(self, op: HashDedup, column: ColumnExpr):
        self.op = op
        self.column = column

    def plan(
        self, dataset: ray.data.Dataset, compute: ComputeStrategy | None = None, **ray_remote_args
    ) -> ray.data.Dataset:
        if not isinstance(dataset, MaterializedDataset) and not isinstance(dataset._logical_plan.dag, SourceOperator):
            raise ValueError(
                f"The filter `{type(self.op).__name__}` can only be used on a materialized dataset or "
                f"the input is a Datasource (e.g. from_pyarrow)."
            )
        return self.op._process(
            text_column=self.column,
            dataset=dataset,
            compute=compute,
            **ray_remote_args,
        )
