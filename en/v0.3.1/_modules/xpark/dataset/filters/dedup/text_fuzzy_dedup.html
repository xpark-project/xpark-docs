def optimal_param(
    threshold: float,
    num_perm: int,
    false_positive_weight: float = 0.5,
    false_negative_weight: float = 0.5,
):
    """
    Compute the optimal `MinHashLSH` parameter that minimizes the weighted sum
    of probabilities of false positive and false negative, taken from
    datasketch.

    :param threshold: float. The threshold for similarity
    :param num_perm: int. The number of permutations
    :param false_positive_weight: float. The weight of false positive
    :param false_negative_weight: float. The weight of false negative
    :return: Tuple[int, int]. The optimal `b` and `r` parameters. The number of
        bands, and the number of rows per band respectively
    """

    def false_positive_probability(th: float, band: int, rows: int):
        """Source: `datasketch.lsh`"""

        def proba(s):
            return 1 - (1 - s ** float(rows)) ** float(band)

        a, _ = integrate.quad(proba, 0.0, th)
        return a

    def false_negative_probability(th: float, band: int, rows: int):
        """Source: `datasketch.lsh`"""

        def proba(s):
            return 1 - (1 - (1 - s ** float(rows)) ** float(band))

        a, _ = integrate.quad(proba, th, 1.0)
        return a

    # object: minimize the weighted FP and FN ratio
    min_error = float("inf")
    opt = (0, 0)
    for b in range(1, num_perm + 1):
        max_r = int(num_perm / b)
        for r in range(1, max_r + 1):
            fp = false_positive_probability(threshold, b, r)
            fn = false_negative_probability(threshold, b, r)
            error = fp * false_positive_weight + fn * false_negative_weight
            if error < min_error:
                min_error = error
                opt = (b, r)
    return opt


def string_array_shingling(
    arr: pa.Array, k: int, regex_patten: str | None = None, cjk=False
) -> Generator[list[str], None, None]:
    """
    Process shingling to leverage PyArrow's compute functions for tokenization.

    This is a generator function that yields shingles for each string one at a time,
    which is more memory-efficient for large datasets.

    Args:
        arr: PyArrow string array containing the input text
        k: Size of each shingle (number of tokens)
        regex_patten: Regex pattern to ignore when computing shingling,
            use utf8_split_whitespace if regex_patten is None
        cjk: Whether to use cjk or not, default False. Only available if regex_patten is None

    Yields:
        List of shingles for each string in the input array
    """
    # Split all strings into tokens at once
    if regex_patten is None:
        if cjk:
            tokens = dedup_utils.utf8_split_mixed(arr)
        else:
            tokens = pc.utf8_split_whitespace(arr)
    else:
        tokens = pc.split_pattern_regex(arr, pattern=regex_patten)

    yield from dedup_utils.gen_shingling_from_array(tokens, k=k)

class TextFuzzyDedup:
    """
    Text fuzzy deduplication using MinHashLSH.

    This class performs fuzzy deduplication on text data by:
    1. Shingling: Breaking text into overlapping token sequences (shingles)
    2. MinHash: Computing compact signatures for each text
    3. LSH (Locality-Sensitive Hashing): Efficiently finding similar texts
    4. Union-Find: Grouping duplicates using a distributed union-find algorithm

    Args:
        tokenize_regex_pattern: Split by space if None else apply the regex to tokenize
        shingling_window_size: Size of shingles (tokens for space, characters for character)
        lowercase: Whether to convert text to lowercase before processing
        cjk: Whether to use CJK split, only available if tokenize_regex_pattern is None
        jaccard_threshold: Jaccard similarity threshold for duplicates
        minhash_num_perm: Number of permutations for MinHash
        minhash_seed: Seed for random number generation (optional)
        lsh_num_bands: Number of bands for LSH (computed if None)
        lsh_num_rows_per_band: Number of rows per band (computed if None)
        union_find_parallel_num: Number of union find to use for parallelization
        union_batch_size: Batch size for union finding
        union_balancing_batch_size: BTS batch size for union finding

    Example:
        >>> from xpark.dataset import read_parquet, TextFuzzyDedup
        >>> from xpark.dataset.expressions import col
        >>>
        >>> # Read parquet files with `dynamic_uid` to generate a unique ID column
        >>> ds = read_parquet("/data/fineweb-edu-sample-10BT", dynamic_uid="uid")
        >>>
        >>> # Apply fuzzy deduplication and save results
        >>> ds.filter(
        ...     TextFuzzyDedup().with_column(uid=col("uid"), text=col("text"))
        ... ).drop_columns("uid").write_parquet("/data/dedup-fineweb-edu-sample-10BT")
    """

    def __init__(
        self,
        tokenize_regex_pattern: str | None = None,
        shingling_window_size: int = 5,
        lowercase: bool = True,
        cjk: bool = True,
        jaccard_threshold: float = 0.7,
        minhash_num_perm: int = 256,
        minhash_seed: int = 42,
        lsh_num_bands: int | None = None,
        lsh_num_rows_per_band: int | None = None,
        union_find_parallel_num: int | None = None,
        union_batch_size: int = 256,
        union_balancing_batch_size: int = 1,
    ):
        self.tokenize_regex_pattern = tokenize_regex_pattern
        self.shingling_window_size = shingling_window_size
        self.lowercase = lowercase
        self.cjk = cjk
        self.jaccard_threshold = jaccard_threshold
        self.num_perm = minhash_num_perm
        self.seed = minhash_seed
        self.num_bands = lsh_num_bands
        self.num_rows_per_band = lsh_num_rows_per_band
        self.union_batch_size = union_batch_size
        self.union_balancing_batch_size = union_balancing_batch_size

        # Calculate optimal LSH parameters if not provided
        if self.num_bands is None or self.num_rows_per_band is None:
            self.num_bands, self.num_rows_per_band = optimal_param(self.jaccard_threshold, self.num_perm)

        # Compute hash ranges and create hash tables
        self.hash_ranges = [
            (i * self.num_rows_per_band, (i + 1) * self.num_rows_per_band) for i in range(self.num_bands)
        ]

        if union_find_parallel_num is None:
            self.union_find_parallel_num = max(int(ray.cluster_resources().get("CPU") / 4), 1)
        else:
            self.union_find_parallel_num = int(union_find_parallel_num)

        assert self.union_find_parallel_num > 0

        dedup_uid = uuid.uuid4().hex[:8]
        logger.info(
            "TextFuzzyDedup[%s] union find parallel num: %s, minhash band num: %s, num rows per band: %s",
            dedup_uid,
            self.union_find_parallel_num,
            self.num_bands,
            self.num_rows_per_band,
        )
        self.union_find_actors = [
            UnionFindActor.options(
                name=UnionFindActorNameTemplate.format(dedup_uid=dedup_uid, index=index), get_if_exists=True
            ).remote(
                dedup_uid=dedup_uid,
                parallel_id=index,
                parallel_num=self.union_find_parallel_num,
                union_batch_size=self.union_batch_size,
                union_balancing_batch_size=self.union_balancing_batch_size,
            )
            for index in range(self.union_find_parallel_num)
        ]

    def _preprocess(
        self,
        uid_column: ColumnExpr,
        text_column: ColumnExpr,
        dataset: ray.data.Dataset,
        compute: ComputeStrategy | None = None,
        **ray_remote_args,
    ) -> None:
        lowercase = self.lowercase
        cjk = self.cjk
        num_perm = self.num_perm
        seed = self.seed
        hash_ranges = self.hash_ranges
        regex_pattern = self.tokenize_regex_pattern
        shingling_window_size = self.shingling_window_size
        union_find_parallel_num = self.union_find_parallel_num
        union_find_actors = self.union_find_actors

        @udf(return_dtype=pa.bool_())
        def _gen_dup_info(_uid_array: pa.ChunkedArray, _text_array: pa.ChunkedArray) -> pa.Array:
            from rensa import RMinHash

            # Simple normalize input text
            if lowercase:
                _text_array = pc.utf8_lower(_text_array)

            # Shingling text for minhash
            shinglings = string_array_shingling(_text_array, shingling_window_size, regex_pattern, cjk)

            # Group band lsh
            lsh_dict = defaultdict(list)
            for uid, s in zip(_uid_array, shinglings):
                m = RMinHash(num_perm=num_perm, seed=seed)
                m.update(s)
                hash_values = m.digest()  # list of uint32
                for i, (start, end) in enumerate(hash_ranges):
                    band_minhash = hash_values[start:end]
                    band_minhash = struct.pack(f"{len(band_minhash) + 1}I", i, *band_minhash)
                    parallel_id = hash_values[start] % union_find_parallel_num
                    lsh_dict[parallel_id].append((band_minhash, uid.as_py()))

            result_refs = [union_find_actors[i].group_uids.remote(p) for i, p in lsh_dict.items()]
            ray.get(result_refs)
            return pa.array(np.zeros(len(_uid_array), dtype=np.bool_))

        # As we filtered out all the data, this write_parquet should be a noop.
        dataset.filter(expr=_gen_dup_info(uid_column, text_column), compute=compute, **ray_remote_args).write_parquet(
            "/tmp"
        )

    def _build_bloom_filters(self):
        logger.info("Redistribution...")
        ray.wait(
            [actor.redistribution.remote() for actor in self.union_find_actors],
            num_returns=len(self.union_find_actors),
            fetch_local=False,
        )

        logger.info("Balancing...")
        count = 1
        while any(ray.get([actor.balanced_union_find.remote() for actor in self.union_find_actors])):
            logger.info("Balancing round {}".format(count))
            ray.wait(
                [actor.communication.remote() for actor in self.union_find_actors],
                num_returns=len(self.union_find_actors),
                fetch_local=False,
            )
            count += 1

        logger.info("Building bloom filters...")
        ray.wait(
            [actor.build_bloom_filter.remote() for actor in self.union_find_actors],
            num_returns=len(self.union_find_actors),
            fetch_local=False,
        )
        logger.info("Bloom filters built.")

    def _postprocess(
        self,
        uid_column: ColumnExpr,
        text_column: ColumnExpr,
        dataset: ray.data.Dataset,
        compute: ComputeStrategy | None = None,
        **ray_remote_args,
    ) -> ray.data.Dataset:
        union_find_parallel_num = self.union_find_parallel_num
        union_find_actors = self.union_find_actors

        @udf(return_dtype=pa.bool_())
        def _filter_dup(_uid_array: pa.ChunkedArray, _text_array: pa.ChunkedArray):
            # Group queries
            group_queries = defaultdict(list)
            for idx, uid in enumerate(_uid_array.to_pylist()):
                parallel_id = uid % union_find_parallel_num
                group_queries[parallel_id].append((uid, idx))

            # Query bloom filters
            result_refs = [
                union_find_actors[parallel_id].query_bloom_filter.remote(query)
                for parallel_id, query in group_queries.items()
            ]
            dup_mask_indexes = ray.get(result_refs)

            # Update mask
            mask = np.ones(len(_text_array), dtype=np.bool_)
            for dup_mask_index in dup_mask_indexes:
                mask[dup_mask_index] = False
            del dup_mask_indexes
            return pa.array(mask)

        ds = dataset.filter(expr=_filter_dup(uid_column, text_column), compute=compute, **ray_remote_args)
        return ds

    def with_column(self, uid: ColumnExpr, text: ColumnExpr) -> DedupOp:
        """Apply the TextFuzzyDedup filter with specified columns.

        Args:
            uid: Column expression for unique identifiers. Should be integer type.
                Use `dynamic_uid` when reading data to auto-generate unique IDs.
            text: Column expression for text content to deduplicate.

        Returns:
            DedupOp: A deduplication operation that can be used with `filter()`.

        Example:
            >>> from xpark.dataset import read_parquet, TextFuzzyDedup
            >>> from xpark.dataset.expressions import col
            >>> ds = read_parquet("data/", dynamic_uid="uid")
            >>> ds.filter(TextFuzzyDedup().with_column(uid=col("uid"), text=col("text")))
        """
        return TextFuzzyDedupOp(self, uid, text)






class TextFuzzyDedupOp(DedupOp):
    def __init__(self, op: TextFuzzyDedup, uid_column: ColumnExpr, text_column: ColumnExpr):
        self.op = op
        self.uid_column = uid_column
        self.text_column = text_column

    def plan(
        self, dataset: ray.data.Dataset, compute: ComputeStrategy | None = None, **ray_remote_args
    ) -> ray.data.Dataset:
        if not isinstance(dataset, MaterializedDataset) and not isinstance(dataset._logical_plan.dag, SourceOperator):
            # TODO(baobliu): We can implement a special optimize rule to push the dedup operation
            #  to the front of the dag to avoid duplicate calculation.
            raise Exception(
                f"The filter `{type(self.op).__name__}` can only be used on a materialized dataset or "
                f"the input is a Datasource (e.g. from_pyarrow)."
            )
        self.op._preprocess(
            uid_column=self.uid_column,
            text_column=self.text_column,
            dataset=dataset,
            compute=compute,
            **ray_remote_args,
        )
        self.op._build_bloom_filters()
        return self.op._postprocess(
            uid_column=self.uid_column,
            text_column=self.text_column,
            dataset=dataset,
            compute=compute,
            **ray_remote_args,
        )
