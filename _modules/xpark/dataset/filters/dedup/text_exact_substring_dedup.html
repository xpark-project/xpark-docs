class ExactDedupActor:
    def __init__(
        self,
        parallel_id: int,
        dedup_uid: int,
        regex_pattern: str | None = None,
        cjk: bool = True,
        min_match_tokens: int = 200,
        union_batch_size: int = 256,
    ):
        self.parallel_id = parallel_id
        self.dedup_uid = dedup_uid
        self.regex_pattern = regex_pattern
        self.cjk = cjk
        self.min_match_tokens = min_match_tokens
        self.union_batch_size = union_batch_size
        self.count = 0
        self.lsh: dict[int, pa.Table | None] = {}

    def update_column_name(self, uid_column_name: str, text_column_name: str):
        self.uid_name = uid_column_name
        self.text_column_name = text_column_name

    def group_data(self, data: list[tuple[int, pa.Table]]):
        self.count += len(data)
        for parent_id, table in data:
            grouped_data = self.lsh.get(parent_id)
            if grouped_data is None:
                self.lsh[parent_id] = [table]
            else:
                # Makes mypy happy.
                assert type(grouped_data) is list
                grouped_data.append(table)
                if len(grouped_data) >= self.union_batch_size:
                    grouped_data[:] = [pa.concat_tables(grouped_data)]

    def _dedup_table(self, table: pa.Table) -> pa.Table:
        tokens = text_tokenize(table[self.text_column_name], regex_patten=self.regex_pattern, cjk=self.cjk)

        table = table.set_column(
            table.schema.get_field_index(self.text_column_name),
            self.text_column_name,
            winnowing.exact_substring_dedup(tokens, table[self.text_column_name], self.min_match_tokens),
        )

        mask = pc.is_valid(table[self.text_column_name])
        mask = pc.and_(mask, pc.not_equal(table[self.text_column_name], ""))
        table = table.filter(mask)

        return table

    def build_pa_table(self) -> pa.Table:
        tables = []
        for _, table_list in self.lsh.items():
            table = pa.concat_tables(table_list)
            if len(table) <= 1:
                tables.append(table)
            else:
                new_table = self._dedup_table(table)
                if len(new_table) >= 1:
                    tables.append(new_table)
        if tables:
            return pa.concat_tables(tables)
        else:
            return pa.table({})

class TextExactSubstringDedup:
    """
    Implements a scalable, distributed exact substring deduplication algorithm using a divide-and-conquer strategy.

    Designed for large-scale text datasets, this class operates on the assumption that duplicated segments constitute
    a small fraction of the total volume. It employs a two-stage process: coarse-grained clustering followed by
    fine-grained deduplication.

    Algorithm Workflow:

    1. Coarse-grained Filtering:
       Generates document fingerprints using x-gram shingling(x is min_match_tokens // 2), Winnowing (Local Min-Hash),
       and Karp-Rabin hashing. Documents sharing identical fingerprints are treated as connected nodes in a graph.

    2. Connectivity Analysis:
       Uses a Distributed Union-Find (BTS) to identify Connected Components (subgraphs) of related documents.
       - Isolated Nodes: Documents with no overlaps are passed through immediately to the main output stream.
       - Connected Components: Documents sharing substrings are routed to distributed Actors for buffering.

    3. Fine-grained Deduplication:
       Within each Actor, a Suffix Array (SA) is constructed for the buffered document cluster.
       Exact substring duplication is performed by analyzing the Longest Common Prefix (LCP) array to identify
       and remove redundant segments.

    Args:
        tokenize_regex_pattern: Split by space if None else apply the regex to tokenize
        min_match_tokens: Minimum number of tokens to match
        cjk: Whether to use CJK split, only available if tokenize_regex_pattern is None
        union_find_parallel_num: Number of union find to use for parallelization
        union_batch_size: Batch size for union finding
        dedup_parallel_num: Number of dedup actors to use for parallelization

    Example:
        >>> from xpark.dataset import read_parquet, TextExactSubstringDedup
        >>> from xpark.dataset.expressions import col
        >>> ds = read_parquet("data/", dynamic_uid="uid")
        >>> dedup_op = TextExactSubstringDedup()
        >>> ds_isolated = ds.filter(dedup_op.with_column(uid=col("uid"), text=col("text")))
        >>> ds_isolated.write_parquet("data_deduped/")
        >>> ds_connected_deduped = dedup_op.collect_sa_deduped_dataset()
        >>> ds_connected_deduped.write_parquet("data_deduped/")

    """

    def __init__(
        self,
        tokenize_regex_pattern: str | None = None,
        cjk: bool = True,
        min_match_tokens: int = 200,
        union_parallel_num: int | None = None,
        dedup_parallel_num: int | None = None,
        union_batch_size: int = 256,
    ):
        self.tokenize_regex_pattern = tokenize_regex_pattern
        self.cjk = cjk
        self.min_match_tokens = min_match_tokens
        self.winnowing_kgram_size = min_match_tokens // 2
        self.union_batch_size = union_batch_size

        if union_parallel_num is None:
            self.parallel_num = max(int(ray.cluster_resources().get("CPU") / 4), 1)
        else:
            self.parallel_num = int(union_parallel_num)

        if dedup_parallel_num is None:
            self.dedup_parallel_num = max(int(ray.cluster_resources().get("CPU")), 1)
        else:
            self.dedup_parallel_num = int(dedup_parallel_num)

        assert self.parallel_num > 0

        dedup_uid = uuid.uuid4().hex[:8]
        logger.info(
            "TextExactSubstringDedup[%s] winnowing hash parallel num: %s",
            dedup_uid,
            self.parallel_num,
        )

        self.union_find_actors = [
            UnionFindActor.options(
                name=UnionFindActorNameTemplate.format(dedup_uid=dedup_uid, index=index), get_if_exists=True
            ).remote(
                dedup_uid=dedup_uid,
                parallel_id=index,
                parallel_num=self.parallel_num,
                union_batch_size=self.union_batch_size,
                union_balancing_batch_size=1,
            )
            for index in range(self.parallel_num)
        ]

        self.exact_dedup_actor = [
            ExactDedupActor.options(
                name=ExactDedupActorNameTemplate.format(
                    dedup_uid=dedup_uid, index=index, union_batch_size=self.union_batch_size
                ),
                get_if_exists=True,
            ).remote(
                parallel_id=index,
                dedup_uid=dedup_uid,
                regex_pattern=self.tokenize_regex_pattern,
                cjk=self.cjk,
                min_match_tokens=self.min_match_tokens,
            )
            for index in range(self.dedup_parallel_num)
        ]

    def _preprocess(
        self,
        uid_column: ColumnExpr,
        text_column: ColumnExpr,
        dataset: ray.data.Dataset,
        compute: ComputeStrategy | None = None,
        **ray_remote_args,
    ) -> None:
        cjk = self.cjk
        regex_pattern = self.tokenize_regex_pattern
        parallel_num = self.parallel_num
        union_find_actors = self.union_find_actors

        @udf(return_dtype=pa.bool_())
        def _gen_dup_info(_uid_array: pa.ChunkedArray, _text_array: pa.ChunkedArray) -> pa.Array:
            tokens = text_tokenize(_text_array, regex_patten=regex_pattern, cjk=cjk)
            hashs_array = winnowing.winnowing(tokens, self.winnowing_kgram_size, self.winnowing_kgram_size + 1)
            lsh_dict = defaultdict(list)

            for uid, hashs in zip(_uid_array, hashs_array):
                for hash in hashs:
                    # TODO(anthonycai) use bytes replace uint64_t
                    hash = hash.as_py()
                    hash_bytes = hash.to_bytes(8, byteorder="big", signed=False)
                    parallel_id = hash % parallel_num
                    lsh_dict[parallel_id].append((hash_bytes, uid.as_py()))

            result_refs = [union_find_actors[i].group_uids.remote(p) for i, p in lsh_dict.items()]
            ray.get(result_refs)

            return pa.array(np.zeros(len(_uid_array), dtype=np.bool_))

        dataset.filter(expr=_gen_dup_info(uid_column, text_column), compute=compute, **ray_remote_args).write_parquet(
            "/tmp"
        )

    def _build_bts(self):
        logger.info("Redistribution...")
        ray.wait(
            [actor.redistribution.remote() for actor in self.union_find_actors],
            num_returns=len(self.union_find_actors),
            fetch_local=False,
        )

        logger.info("Balancing...")
        count = 1
        while any(ray.get([actor.balanced_union_find.remote() for actor in self.union_find_actors])):
            logger.info("Balancing round {}".format(count))
            ray.wait(
                [actor.communication.remote() for actor in self.union_find_actors],
                num_returns=len(self.union_find_actors),
                fetch_local=False,
            )
            count += 1

        logger.info("Building bloom filters...")
        ray.wait(
            [actor.build_bloom_filter.remote(with_parent=True) for actor in self.union_find_actors],
            num_returns=len(self.union_find_actors),
            fetch_local=True,
        )
        logger.info("Bloom filters built.")

    def _postprocess(
        self,
        uid_column: ColumnExpr,
        text_column: ColumnExpr,
        dataset: ray.data.Dataset,
        compute: ComputeStrategy | None = None,
        **ray_remote_args,
    ):
        parallel_num = self.parallel_num
        union_find_actors = self.union_find_actors
        dedup_parallel_num = self.dedup_parallel_num

        def _filter_dup_and_record(table: pa.Table, uid_column_str: str, text_column_name: str) -> pa.Table:
            _uid_array = table[uid_column_str]
            _text_array = table[text_column_name]

            # Group queries
            group_queries = defaultdict(list)
            for idx, uid in enumerate(_uid_array.to_pylist()):
                parallel_id = uid % parallel_num
                group_queries[parallel_id].append((uid, idx))

            # Query bloom filter
            result_refs = [
                union_find_actors[parallel_id].query_bloom_filter_and_get_parent_id.remote(query)
                for parallel_id, query in group_queries.items()
            ]
            dup_mask_indexes = ray.get(result_refs)

            dedup_queries = defaultdict(list)
            mask = np.ones(len(_text_array), dtype=np.bool_)
            for dup_mask_index, parent_id in list(chain.from_iterable(dup_mask_indexes)):
                parallel_id = parent_id % dedup_parallel_num
                mask[dup_mask_index] = False
                dedup_queries[parallel_id].append((parent_id, table.slice(dup_mask_index, 1)))

            del dup_mask_indexes
            ray.wait(
                [
                    actor.group_data.remote(dedup_queries[parallel_id])
                    for parallel_id, actor in enumerate(self.exact_dedup_actor)
                ],
                num_returns=len(self.exact_dedup_actor),
                fetch_local=False,
            )

            mask_array = pa.array(mask)
            filtered_table = table.filter(mask_array)

            return filtered_table

        ds = dataset.map_batches(
            fn=_filter_dup_and_record,
            fn_args=(uid_column.name, text_column.name),
            compute=compute,
            batch_format="pyarrow",
            **ray_remote_args,
        )

        return ds

    def collect_sa_deduped_dataset(self) -> Dataset:
        """Collects the 'Connected Components' dataset from distributed actors
        following Suffix Array (SA) deduplication. This method retrieves documents that
        were flagged as potential duplicates and offloaded to actors for exact deduplication."""
        from xpark.dataset.read_api import from_arrow_refs

        ray.wait(
            [actor.__ray_terminate__.remote() for actor in self.union_find_actors],
            num_returns=len(self.union_find_actors),
        )
        tables_refs = [actor.build_pa_table.remote() for actor in self.exact_dedup_actor]
        return from_arrow_refs(tables_refs)

    def with_column(self, uid: ColumnExpr, text: ColumnExpr) -> TextExactSubstringDedupOp:
        """Apply the TextExactSubstringDedup filter with specified columns.

        Args:
            uid: Column expression for unique identifiers. Should be integer type.
                Use `dynamic_uid` when reading data to auto-generate unique IDs.
            text: Column expression for text content to deduplicate.

        Returns:
            DedupOp: A deduplication operation that can be used with `filter()`.

        Example:
            >>> from xpark.dataset import read_parquet, TextExactSubstringDedup
            >>> from xpark.dataset.expressions import col
            >>> ds = read_parquet("data/", dynamic_uid="uid")
            >>> dedup_op = TextExactSubstringDedup()
            >>> ds_isolated = ds.filter(dedup_op.with_column(uid=col("uid"), text=col("text")))
            >>> ds_isolated.write_parquet("data_deduped/")
            >>> ds_connected_deduped = dedup_op.collect_sa_deduped_dataset()
            >>> ds_connected_deduped.write_parquet("data_deduped/")
        """
        for actor in self.exact_dedup_actor:
            ray.get(actor.update_column_name.remote(uid.name, text.name))

        return TextExactSubstringDedupOp(self, uid, text)






class TextExactSubstringDedupOp(DedupOp):
    def __init__(self, op: TextExactSubstringDedup, uid_column: ColumnExpr, text_column: ColumnExpr):
        self.op = op
        self.uid_column = uid_column
        self.text_column = text_column

    def plan(
        self, dataset: ray.data.Dataset, compute: ComputeStrategy | None = None, **ray_remote_args
    ) -> ray.data.Dataset:
        if not isinstance(dataset, MaterializedDataset) and not isinstance(dataset._logical_plan.dag, SourceOperator):
            raise Exception(
                f"The filter `{type(self.op).__name__}` can only be used on a materialized dataset or "
                f"the input is a Datasource (e.g. from_pyarrow)."
            )
        self.op._preprocess(
            uid_column=self.uid_column,
            text_column=self.text_column,
            dataset=dataset,
            compute=compute,
            **ray_remote_args,
        )

        self.op._build_bts()

        dataset = self.op._postprocess(
            uid_column=self.uid_column,
            text_column=self.text_column,
            dataset=dataset,
            compute=compute,
            **ray_remote_args,
        )

        return dataset

    def collect_sa_deduped_dataset(
        self,
    ) -> Dataset:
        return self.op.collect_sa_deduped_dataset()
